284 {

290  float distY = (p1.getY() - p2.getY());
291 
292  float len = sqrt( (distX * distX ) + (distY * distY ) );
293  float dot = (dotProduct( p3, p1, p2 )) / (len*len);
294

297  return Point(closestX, closestY);
